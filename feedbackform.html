                                (function() { // ================= CONFIGURATION ================= const TARGET_TEXT = "Excellent"; // The rating you want to pick const CHECK_INTERVAL = 30; // How often to check (ms) const TIMEOUT_LIMIT = 3000; // Max wait time per box before skipping (ms) // ================= STATE MANAGEMENT ================= let isRunning = false; let stopRequested = false; // ================= GUI CREATION ================= function createUI() { // Remove existing panel if any const existing = document.getElementById('automation-panel'); if (existing) existing.remove(); const panel = document.createElement('div'); panel.id = 'automation-panel'; panel.style.cssText = ` position: fixed; top: 20px; right: 20px; z-index: 99999; background: #222; color: #fff; padding: 15px; border-radius: 8px; font-family: sans-serif; box-shadow: 0 4px 15px rgba(0,0,0,0.5); width: 250px; `; panel.innerHTML = `

                                Feedback Filler

                                Ready to start.
                                `; document.body.appendChild(panel); document.getElementById('btn-start').onclick = startAutomation; document.getElementById('btn-stop').onclick = () => { stopRequested = true; updateStatus("Stopping...", "orange"); }; } // ================= HELPER FUNCTIONS ================= const wait = (ms) => new Promise(r => setTimeout(r, ms)); function updateStatus(text, color = "#aaa") { const el = document.getElementById('status-text'); if (el) { el.innerText = text; el.style.color = color; } } function updateProgress(current, total) { const percent = (current / total) * 100; const bar = document.getElementById('progress-bar-fill'); if (bar) bar.style.width = `${percent}%`; } // A smart waiter that looks for the popup to appear function waitForDropdownToOpen() { return new Promise((resolve, reject) => { const startTime = Date.now(); const interval = setInterval(() => { const popup = document.getElementById('select2-drop'); // Check if popup exists and is visible (display block) if (popup && popup.style.display !== 'none' && popup.offsetHeight > 0) { clearInterval(interval); resolve(popup); } if (Date.now() - startTime > TIMEOUT_LIMIT) { clearInterval(interval); reject("Timeout waiting for dropdown"); } }, CHECK_INTERVAL); }); } // ================= MAIN LOGIC ================= async function startAutomation() { if (isRunning) return; isRunning = true; stopRequested = false; // Find all Target Dropdowns const forms = document.querySelectorAll('.select2-container .select2-choice'); const total = forms.length; updateStatus(`Found ${total} dropdowns. Starting...`, "#fff"); for (let i = 0; i < total; i++) { if (stopRequested) { updateStatus(" Stopped by user.", "red"); break; } updateProgress(i + 1, total); const button = forms[i]; // 1. Scroll into view (helps with lazy loading) button.scrollIntoView({ block: 'center', behavior: 'instant' }); // 2. Open Dropdown // We use mousedown as established previously button.dispatchEvent(new MouseEvent('mousedown', { bubbles: true })); try { // 3. Wait for the global popup to appear (Dynamic Waiting) const popup = await waitForDropdownToOpen(); // 4. Find the option inside the popup // Note: The popup ID is always 'select2-drop' const options = popup.querySelectorAll('.select2-result-label'); let found = false; for (const opt of options) { if (opt.textContent.trim() === TARGET_TEXT) { // Click it opt.dispatchEvent(new MouseEvent('mouseup', { bubbles: true, view: window })); // Fallback click opt.click(); found = true; break; } } if (!found) { console.warn(`Option "${TARGET_TEXT}" not found in dropdown #${i}`); // Close it to reset state button.dispatchEvent(new MouseEvent('mousedown', { bubbles: true })); } } catch (error) { console.error(`Error on dropdown #${i}:`, error); updateStatus(`Error on #${i}. Skipping...`, "orange"); // Attempt to close if stuck open button.dispatchEvent(new MouseEvent('mousedown', { bubbles: true })); } // Tiny cooldown to let the DOM settle before the next one // This prevents the "double open" bug await wait(50); } if (!stopRequested) { updateStatus(" All done!", "#28a745"); } isRunning = false; } // Initialize createUI(); })();
